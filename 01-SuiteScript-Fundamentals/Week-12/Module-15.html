Key Patterns

// Load a record
var rec = record.load({ type: 'customer', id: 123 });

// Create a search
var search = search.create({
    type: 'salesorder',
    filters: [['status', 'anyof', 'SalesOrd:B']],
    columns: ['tranid', 'total']
});

// Process results
search.run().each(function(result) {
    log.debug('Order', result.getValue('tranid'));
    return true;
});
